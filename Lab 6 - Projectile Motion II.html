import numpy as np
import matplotlib.pyplot as plt

# (a) Define some helpful variables:
#     (dt - time step, g - acceleration due to gravity)
dt = 0.001    # s
g  = -9.81    # m/s^2


#(b) Choose some initial conditions for position and velocity. 
#    (angle - launch angle, v - launch speed, h - launch height)
angle = 45    # degrees
v     = 30    # m/s 
h     = 30    # m

# (c) Initialize lists for time, x, y, vx , and vy 
time = [0]    # s
x    = [0]    # m
y    = [h]    # m
vx   = [v * np.cos(np.radians(angle))] # m/s
vy   = [v * np.sin(np.radians(angle))] # m/s

# (d) Construct a while loop to iterate over time and calculate future values
#     of position and velocity using equations (6.7) and (6.8).
while (y[-1] > 0.0 or vy[-1] > 0.0):
    x    .append(x[-1] + vx[-1] * dt)
    y    .append(y[-1] + vy[-1] * dt)
    vx   .append(vx[-1])
    vy   .append(vy[-1] + g * dt)
    time .append(time[-1] + dt)

# box up the plotting code into a function so that we can reuse it.
def plot_nice(y , y_name,  x, x_name):
    plt.title(y_name + ' vs ' + x_name)
    plt.xlabel(x_name)
    plt.ylabel(y_name)
    plt.plot(x, y)
    plt.show() 
    
# (e) Plot vx and vy vs time.
plot_nice(vx,"Velocity in the X Direction", time,"Time")
plot_nice(vy,"Velocity in the Y Direction", time,"Time")

# (f) Plot x and y vs time..
plot_nice(x,"X Position", time,"Time")
plot_nice(y,"Y Position", time,"Time")

# (g) Plot y vs x. This is called a trajectory plot.
plot_nice(y,"Y Position", x,"X Position")
